###### True & True
```js
let pao = true
let queijo = true
// AND &&
pao && queijo // Resultado é true
// OR || 
pao || queijo // Resultado é true
// NOT !
!pao // Resultado é falso
!queijo // Resultado é falso
```
###### True & False
```js
let pao = true
let queijo = false
// AND &&
pao && queijo // Resultado é false
// OR || 
pao || queijo // Resultado é true
// NOT !
!pao // Resultado é falso
!queijo // Resultado é true
```
###### False & False
```js
let pao = false
let queijo = false
// AND &&
pao && queijo // Resultado é falso
// OR || 
pao || queijo // Resultado é falso
// NOT !
!pao // Resultado é true
!queijo // Resultado é true
```
